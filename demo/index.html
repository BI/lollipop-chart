Lollipop Demo

(with the three scale configuration use cases)
Let the chart make scale calculations based on the data
					
// Example code: 
var lollipopChartBasic = LollipopChart('#lollipop-chart-basic');
var gradesPerName = [
    {name: 'Bobby', value: 80, comparisonValue: 80},
    {name: 'Jane', value: 79, comparisonValue: 68},
    {name: 'Fred', value: 70, comparisonValue: 78},
    {name: 'Alan', value: 75, comparisonValue: 64},
    {name: 'Jeeves', value: 75, comparisonValue: 85},
    {name: 'Mary', value: 67, comparisonValue: 70},
    {name: 'Marsha', value: 60, comparisonValue: 68},
    {name: 'Andy', value: 100, comparisonValue: 100},
  ];
lollipopChartBasic
	.render(gradesPerName);
Set a specific scale to use
					
// Example code: 
var lollipopChartWithScale = LollipopChart('#lollipop-chart-scale');
var gradesPerName = [
    {name: 'Bobby', value: 80, comparisonValue: 80},
    {name: 'Jane', value: 79, comparisonValue: 68},
    {name: 'Fred', value: 70, comparisonValue: 78},
    {name: 'Alan', value: 75, comparisonValue: 64},
    {name: 'Jeeves', value: 75, comparisonValue: 85},
    {name: 'Mary', value: 67, comparisonValue: 70},
    {name: 'Marsha', value: 60, comparisonValue: 68},
    {name: 'Andy', value: 100, comparisonValue: 100},
  ];
lollipopChartBasic
	.yScale(d3.scale.linear().domain([50, 100]))
	.render(gradesPerName);
Pass the chart data that has individual scales for each data
